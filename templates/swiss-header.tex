% Minimal Swiss report style for XeLaTeX export.
% Colours (ink, muted, heading, posterbg, posterfg, backpagebg) are defined
% in report.tex template via YAML variables — do not redefine them here.

\color{ink}

% Math formulas are handled by Pandoc's unicode-math pipeline (XeLaTeX).
\usepackage{textcomp}

% No headers; only page number at bottom-right.
\usepackage{fancyhdr}
\usepackage{tocloft}
\pagestyle{fancy}
\fancyhf{}
\setlength{\footskip}{100pt}
\fancyfoot[R]{\color{ink}\normalsize\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
% Chapter-opening pages also use plain style — keep consistent
\fancypagestyle{plain}{\fancyhf{}\setlength{\footskip}{90pt}\fancyfoot[R]{\color{ink}\Large\thepage}\renewcommand{\headrulewidth}{0pt}\renewcommand{\footrulewidth}{0pt}}
% Customize TOC: no dots, keep page numbers
\renewcommand{\cftdot}{}
\renewcommand{\cftchappagefont}{\normalfont}

% Clean hierarchy, left aligned, no separator lines.
\setcounter{secnumdepth}{-1}
\setcounter{tocdepth}{3}
\usepackage{titlesec}

% Treat chapters as straight headings and never force double-page breaks.
\titleclass{\chapter}{straight}
\let\cleardoublepage\clearpage

\titleformat{\chapter}[display]
  {\raggedright\normalfont\bfseries\color{heading}}
  {}
  {0pt}
  {\fontsize{40pt}{40pt}\selectfont}

\titlespacing*{\chapter}{0pt}{0pt}{24pt}

\titleformat{\section}[block]
  {\raggedright\normalfont\bfseries\fontsize{18pt}{22pt}\selectfont\color{heading}}
  {\thesection}
  {10pt}
  {}

\titlespacing*{\section}{0pt}{18pt}{10pt}

\titleformat{\subsection}[block]
  {\raggedright\normalfont\bfseries\fontsize{13pt}{16pt}\selectfont\color{heading}}
  {\thesubsection}
  {8pt}
  {}

\titlespacing*{\subsection}{0pt}{14pt}{8pt}

\titleformat{\subsubsection}[runin]
  {\normalfont\bfseries\fontsize{11pt}{13pt}\selectfont\color{heading}}
  {\thesubsubsection}
  {6pt}
  {}
  [.] 

\titlespacing*{\subsubsection}{0pt}{10pt}{8pt}

% Body text: tighter spacing to minimize blank pages
\setlength{\parindent}{0pt}
\setlength{\parskip}{6pt plus 1pt minus 0.5pt}
\raggedbottom

% Simple quote style without decorative rules.
\renewenvironment{quote}
  {\list{} {\leftmargin=1.2em\rightmargin=1.2em}\item\relax\itshape\color{muted}}
  {\endlist}

% Hyperlink colors, guarded if hyperref is missing.
\providecommand{\hypersetup}[1]{}
\hypersetup{colorlinks=true,urlcolor=ink,linkcolor=ink,citecolor=ink}

% Title page and TOC have no footer; content starts at page 1 after the TOC.
\AtBeginDocument{%
  \pagenumbering{gobble}%
  \let\frontmatter\relax%
  \let\mainmatter\relax%
  \providecommand{\horizontalrule}{}%
  \renewcommand{\horizontalrule}{}%
  \pagestyle{empty}%
}
