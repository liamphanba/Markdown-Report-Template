% ---- Colors ------------------------------------------------
\usepackage{xcolor}
\definecolor{ink}{HTML}{111111}

% ---- Heading format (titlesec) ----------------------------
% Disable automatic section numbering
\setcounter{secnumdepth}{0}

\usepackage{titlesec}

% section -> H1: 25.63 pt bold, thick rule below
\titleformat{\section}[block]
  {\fontsize{25.63pt}{30.75pt}\selectfont\bfseries\color{ink}}
  {}
  {0pt}
  {}
  [{\vspace{4pt}\textcolor{ink}{\hrule height 2.5pt}}\vspace{15.75pt}]

\titlespacing*{\section}{0pt}{0pt}{0pt}

% subsection -> H2: 16.41 pt bold, thin rule above
\titleformat{\subsection}[block]
  {\fontsize{16.41pt}{19.69pt}\selectfont\bfseries\color{ink}}
  {}
  {0pt}
  {\textcolor{ruleline}{\hrule height 0.4pt}\vspace{10pt}}
  {}

\titlespacing*{\subsection}{0pt}{47.25pt}{0pt}

% subsubsection -> H3: 13.13 pt bold uppercase
\titleformat{\subsubsection}[block]
  {\fontsize{13.13pt}{15.75pt}\selectfont\bfseries\color{ink}}
  {}
  {0pt}
  {}
  {}
